// HttpConnector - WinHTTP Example #include <windows.h> #include <winhttp.h> #include <string> #include <vector> class HttpConnector { private: HINTERNET hSession; HINTERNET hConnect; HINTERNET hRequest; bool IsAllowedCommand(const std::wstring& command) { static const std::vector<std::wstring> allowed = { L"whoami", L"hostname", L"ipconfig", L"systeminfo", L"dir", L"echo", L"ping" }; for (const auto& cmd : allowed) { if (command == cmd || command.rfind(cmd + L" ", 0) == 0) { return true; } } return false; } public: HttpConnector() : hSession(NULL), hConnect(NULL), hRequest(NULL) {} bool Initialize(const std::wstring& server, int port = 80) { hSession = WinHttpOpen(L"HttpConnector/1.0", WINHTTP_ACCESS_TYPE_DEFAULT_PROXY, WINHTTP_NO_PROXY_NAME, WINHTTP_NO_PROXY_BYPASS, 0); if (!hSession) return false; hConnect = WinHttpConnect(hSession, server.c_str(), port, 0); return hConnect != NULL; } bool SendRequest(const std::wstring& path, const std::wstring& method = L"GET") { hRequest = WinHttpOpenRequest(hConnect, method.c_str(), path.c_str(), NULL, WINHTTP_NO_REFERER, WINHTTP_DEFAULT_ACCEPT_TYPES, 0); if (!hRequest) return false; return WinHttpSendRequest(hRequest, WINHTTP_NO_ADDITIONAL_HEADERS, 0, WINHTTP_NO_REQUEST_DATA, 0, 0, 0); } std::string GetResponse() { if (!WinHttpReceiveResponse(hRequest, NULL)) return ""; DWORD dwSize = 0; std::string response; char buffer[4096]; while (WinHttpReadData(hRequest, buffer, sizeof(buffer)-1, &dwSize) && dwSize > 0) { buffer[dwSize] = '\0'; response += buffer; } return response; } ~HttpConnector() { if (hRequest) WinHttpCloseHandle(hRequest); if (hConnect) WinHttpCloseHandle(hConnect); if (hSession) WinHttpCloseHandle(hSession); } };
